SELECT 
DB_NAME(A.DATABASE_ID) AS [DATABASE NAME], 
A.NAME AS [LOGICAL NAME],
A.PHYSICAL_NAME AS [CURRENT FILE LOCATION],
CASE B.cmptlevel
WHEN 80 THEN 2000
WHEN 90 THEN 2005
WHEN 100 THEN 2008
WHEN 110 THEN 2012
END AS [CMPTLEVEL],
B.crdate [DATE CREATED]
FROM SYS.MASTER_FILES A
INNER JOIN SYSDATABASES B 
ON (A.DATABASE_ID = B.DBID)
WHERE A.DATABASE_ID>4
AND A.PHYSICAL_NAME NOT LIKE '%.LDF%'
--AND B.status=65544
ORDER BY 
DB_NAME(A.DATABASE_ID)
